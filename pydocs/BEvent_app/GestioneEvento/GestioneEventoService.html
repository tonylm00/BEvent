 14def is_valid_data(data):
 15    """
 16    Funzione per verificare che la data inserita sia valida. Viene quindi controllato che la data sia > alla data
 17    odierna.
 18
 19    :param data: (str) data inserita dall'utente nella pagina SceltaEventoDaCreare.html
 20
 21    :return: True se la data è corretta, False altrimenti
 22    """
 23    date_format_regex = re.compile(r'^\d{2}-\d{2}-\d{4}$')
 24
 25    if not date_format_regex.match(data):
 26        return False, "Formato data non corretto. Utilizzare il formato dd-mm-yyyy."
 27
 28    try:
 29        datetime_data = datetime.strptime(data, '%d-%m-%Y')
 30
 31        data_odierna = datetime.now()
 32
 33        if datetime_data > data_odierna:
 34            return True, "La data è corretta nel formato ed è anche valida."
 35        else:
 36            return False, "La data non è valida poichè è precedente alla data odierna."
 37
 38    except ValueError:
 39        return False, "Errore nella conversione della data."

 42def get_fornitori_disponibli(data_richiesta):
 43    """
 44    Funzione che ottiene tutti i fornitori che sono disponibili in una determinata data, prendendola dal database.
 45    Usa una pipeline per verificare quali sono disponibili.
 46
 47    :param data_richiesta: (str) stringa che indica la data in cui si vuole creare un evento
 48    :return: lista di oggetti di tipo Fornitore, ovvero i fornitori disponibli
 49    """
 50
 51    pipeline = [
 52        {"$match": {"Ruolo": "3"}},
 53        {
 54            "$lookup": {
 55                "from": "Evento",
 56                "let": {"fornitore_id_str": {"$toString": "$_id"}},
 57                "pipeline": [
 58                    {
 59                        "$unwind": "$Evento.fornitori_associati"
 60                    },
 61                    {
 62                        "$match": {
 63                            "$expr": {
 64                                "$and": [
 65                                    {"$eq": ["$$fornitore_id_str", "$Evento.fornitori_associati"]},
 66                                    {"$eq": ["$Evento.Data", data_richiesta]},
 67                                    {"$eq": ["$Evento.isPagato", True]}
 68                                ]
 69                            }
 70                        }
 71                    }
 72                ],
 73                "as": "eventi_associati"
 74            }
 75        },
 76        {
 77            "$addFields": {
 78                "eventiPrenotati": {"$size": "$eventi_associati"},
 79                "EventiMassimiGiornaliero": "$Fornitore.EventiMassimiGiornaliero"
 80            }
 81        },
 82        {
 83            "$match": {
 84                "$expr": {"$lt": ["$eventiPrenotati", "$EventiMassimiGiornaliero"]}
 85            }
 86        }
 87    ]
 88
 89    fornitori_disponibili = list(db.Utente.aggregate(pipeline))
 90    lista_fornitori = []
 91
 92    for data in fornitori_disponibili:
 93        fornitore = Fornitore(data, data)
 94        lista_fornitori.append(fornitore)
 95
 96    return lista_fornitori

 99def get_servizi(data_richiesta):
100    """
101    Funzione che ottiene dal database tutti i servizi che si possono prenotare in una determinata data, poichè alcuni
102    potrebbero essere impegnati in un evento.
103    -Vengono prese dal databse le collezioni Servizio Offerto e Evento
104    -Viene controllata la versione dei servizi in modo da prendere i servizi la cui versione corrisponde all'ultima
105    rilasciata dal fornitore
106    -Usando la lista di servizi precedentemente filtrati, essi vengono nuovamente filtrati per prendere solo i servizi
107    che non sono prenotati in un evento nella data inserita dall'organizzatore.
108
109    :param data_richiesta: (str) stringa che indica la data in cui si vuole creare un evento
110
111    :return: lista di oggetti di tipo Servizio Offerto, ovvero i servizi disponibli
112    """
113
114    servizi_collection = db['Servizio Offerto']
115    eventi_collection = db['Evento']
116
117    servizi_data = list(
118        servizi_collection.find({'$or': [{'isCurrentVersion': None}, {'isCurrentVersion': {'$exists': False}}]}))
119
120    lista_servizi = []
121
122    for data in servizi_data:
123        servizio = ServizioOfferto(data)
124        evento_associato = eventi_collection.find_one({
125            'servizi_associati': {'$in': [str(servizio._id)]},
126            'Data': data_richiesta,
127            '$or': [
128                {'Ruolo': '2', 'isPagato': True},
129                {'Ruolo': '1'}
130            ]
131        })
132
133        if not evento_associato:
134            lista_servizi.append(servizio)
135
136    return lista_servizi

251def filtro_ricerca(ricerca, data):
252    """
253    Funzione per ottenere dal database la lista di fornitori e servizi la cui descrizione o nome contiene la parola
254    inserita dall'organizzatore.
255    -Vengono prese le liste di fornitori e servizi disponibli nella data indicata dall'organizzatore.
256    -Vengono filtrati i fornitori cercando la parola sia nel nome che nella descrizione
257    -In base alla lista di fornitori selezionati vengono presi i servizi associati.
258
259    :param ricerca: (str) stringa che indica la parola da ricercare scelta dall'organizzatore
260    :param data: (str) stringa che indica la data nella quale si vuole fare l'evento
261
262    :return: due liste filtrate di oggetti: servizi_filtrati (lista di oggetti di tipo Servizio Offerto) e
263    fornitori_filtrati (lista di oggetti di tipo Fornitore)
264
265   """
266    servizi = get_servizi(data)
267    fornitori_non_filtrati = get_fornitori_disponibli(data)
268    servizi_non_filtrati = filtrare_servizi_per_fornitore(servizi, fornitori_non_filtrati)
269
270    fornitori_filtrati_nome = [fornitore for fornitore in fornitori_non_filtrati if
271                               ricerca.lower() in fornitore.nome_utente.lower()
272                               ]
273
274    fornitori_filtrati_desctizione = [fornitore for fornitore in fornitori_non_filtrati if ricerca.lower() in
275                                      fornitore.descrizione.lower()]
276
277    fornitori_filtrati = None
278    if fornitori_filtrati_desctizione and fornitori_filtrati_nome:
279        fornitori_unici = {}
280
281        for fornitore in fornitori_filtrati_nome + fornitori_filtrati_desctizione:
282            fornitori_unici[fornitore.id] = fornitore
283
284        fornitori_filtrati = list(fornitori_unici.values())
285    elif fornitori_filtrati_desctizione:
286        fornitori_filtrati = fornitori_filtrati_desctizione
287    elif fornitori_filtrati_nome:
288        fornitori_filtrati = fornitori_filtrati_nome
289
290    if fornitori_filtrati:
291        servizi_filtrati = filtrare_servizi_per_fornitore(servizi_non_filtrati, fornitori_filtrati)
292    else:
293        servizi_filtrati = None
294
295    return servizi_filtrati, fornitori_filtrati

462    """
463    Funzione per salvare l'evento privato nel database che crea il documento da inserire mettendo come valore dei campi
464    i dati passati come parametri.
465
466    :param    lista_servizi:(list) lista di oggetti servizio offerto
467    :param    lista_fornitori: (list) lista di oggetti fornitore
468    :param    tipo_evento: (str) stringa che rappresenta il tipo di evento che si sta creando
469    :param    data_evento: (str) stringa che rappresenta la data nella quale si svolgerà l'evento
470    :param    n_invitati: (str) stringa che rappresenta il numero di invitati previsti
471    :param    nome_festeggiato: (str) stringa che rappresenta il nome del festeggiato
472    :param    descrizione: (str) stringa che rappresenta una descrizione dell'evento
473    :param    is_pagato: (bool) booleano che indica se l'evento è stato pagato o meno
474    :param    ruolo: (str) stringa che indica se l'evento è pubblico o privato
